// Example: File System MCP Server import { Server } from '@modelcontextprotocol/sdk/server/index.js'; import { readFile, readdir } from 'fs/promises'; const server = new Server({ name: 'filesystem-server', version: '1.0.0' }); server.setRequestHandler('resources/read', async (request) => { const { uri } = request.params; const filePath = uri.replace('file://', ''); try { const content = await readFile(filePath, 'utf-8'); return { contents: [{ uri, mimeType: 'text/plain', text: content }] }; } catch (error) { throw new Error(`Failed to read file: ${error.message}`); } });
